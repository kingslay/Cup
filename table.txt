CREATE TABLE user_info
(
    accountid INT PRIMARY KEY NOT NULL,
    username VARCHAR(50) NOT NULL,
    password VARCHAR(50) NOT NULL,
    birthday CHAR(14),
    nickname VARCHAR(50),
    avatar VARCHAR(100),
    height DOUBLE,
    city VARCHAR(20)
);
CREATE UNIQUE INDEX unique_id ON user_info (accountid);
CREATE UNIQUE INDEX unique_username ON user_info (username);
