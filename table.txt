CREATE TABLE user_info
(
    username VARCHAR(50),
    password VARCHAR(50),
    birthday CHAR(14),
    nickname VARCHAR(50),
    avatar VARCHAR(100),
    height DOUBLE,
    city VARCHAR(20),
    accountid INT PRIMARY KEY NOT NULL AUTO_INCREMENT,
    phone CHAR(11),
    sex CHAR(2),
    scene VARCHAR(128),
    constitution VARCHAR(128),
    weight DOUBLE
);
CREATE UNIQUE INDEX unique_accountid ON user_info (accountid);
CREATE UNIQUE INDEX unique_phone ON user_info (phone);
CREATE UNIQUE INDEX unique_username ON user_info (username);
